def bsgs(a, b, m): # a ** x = b (mod m)
    from math import isqrt
    a %= m
    b %= m
    if b == 1:
        return 0
    n = isqrt(m) + 1
    value = {}
    an = pow(a, n, m)
    cur = b
    for q in range(n):
        value[cur] = q
        cur = cur * a % m
    cur = an
    for p in range(1, n + 2):
        if cur in value:
            return p * n - value[cur]
        cur = cur * an % m
    return -1
